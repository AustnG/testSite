if ... else

let state = 'KY';
let taxPercent;

if (state === 'KY') {
  taxPercent = 6;
}
else if (state === 'NY'){
  taxPercent = 8.875;
}
else{
  taxPercent = 0;
}
console.log(taxPercent); // 6

switch & case

let state = 'TX';

switch(state) {
  case 'NY':
    console.log('New York');
    break; // stops code if condition is met
  case 'TX':
    console.log('Texas');
    break;
  default:
    console.log('Unknown');
    break;
} // result = 'TX' becomes 'Texas'

Looping with for()

for (let i = 0; i < 3; i++) {
  console.log(i);
} // result = 0 1 2

Looping with while()

let count = 1;
while (count < 5) {
  console.log(count);
  count++;
} // result = 1 2 3 4
