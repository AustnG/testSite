JavaScript: Types & Arrays

String & Number Types

let value = "Model";
console.log( typeof(value) ); // = String

let value = 99;
console.log( typeof(value) ); // = Number

let value = 4.1 + 4.3;
console.log(value); // = 8.399999

let value = 10 / 0;
console.log(value); // = Infinity

let value = -10 / 0;
console.log(value); // = -Infinity

let value = 0 / 0;
console.log(value); // = NaN (Not a Number)
